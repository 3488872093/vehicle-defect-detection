1. 创建激活虚拟环境
conda create -n yolo python=3.9
conda activate yolo
2. 安装Pytorch框架
pip3 install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cu118
3. 安装依赖包
cd {YOLOSHOW程序所在的路径}
pip install -r requirements.txt -i https://pypi.tuna.tsinghua.edu.cn/simple
4. 运行程序
python main.py
